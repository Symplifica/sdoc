<script type="text/javascript" charset="utf-8">
$(function() {
  $('#links').hide();
  var panel = new Searchdoc.Panel($('#panel'), search_data, tree, top.frames[1]);
  $('#search').focus();

  var s = window.parent.location.search.match(/\?q=([^&]+)/);
  if (s) {
    s = decodeURIComponent(s[1]).replace(/\+/g, ' ');
    if (s.length > 0) {
      $('#search').val(s);
      panel.search(s, true);
    }
  }
})
</script>
